### Filter Logic Pseudocode

```javascript
// Filter event by tag
function getEventsForFilter(filterName) {
  return timelineEvents.filter(event =>
    event.tags.includes(filterName)
  );
}

// Get event count for filter
function getFilterCount(filterName) {
  return getEventsForFilter(filterName).length;
}

// Validate filter structure
function validateFilters() {
  const requiredFilters = [
    'overview',
    'family-status-discrimination',
    'disability-discrimination',
    'just-cause-failure',
    'bad-faith-conduct',
    'post-termination-misconduct',
    'ohs-retaliation'
  ];

  const counts = {};
  requiredFilters.forEach(filter => {
    counts[filter] = getFilterCount(filter);
  });

  return counts;
}
```
